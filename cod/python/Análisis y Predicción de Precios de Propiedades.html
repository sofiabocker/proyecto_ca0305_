def imputar_por_grupo(df, colums_group, colum_imputar, metodo='mean'):
    # metodo 'max', 'min', 'mean', 'median'
    group_stats = df.groupby(colums_group)[colum_imputar].transform(metodo)
    df[colum_imputar] = np.where(df[colum_imputar].isna(), group_stats, df[colum_imputar])
    return df

# Función que prepara la base de datos para aplicar modelos
# Divide la base de datos en set train y test, convierte las variables categóricas en dummies y escala todas las variables
def preparar_data(df, variable_objetivo, test_size, random_state = None):
    """
    Prepara los datos para el entrenamiento de un modelo dividiendo el conjunto de datos en
    conjuntos de entrenamiento y prueba, y realizando el preprocesamiento necesario.
    
    Parameters:
    -----------
    df : pandas.DataFrame
        El DataFrame que contiene los datos a utilizar.
    
    variable_objetivo : str
        El nombre de la columna en `df` que corresponde a la variable objetivo (dependiente).
    
    test_size : float
        Proporción del conjunto de datos que se destinará a la prueba. Debe estar entre 0 y 1.
    
    random_state : int, opcional
        Semilla para el generador de números aleatorios utilizado en la división de los datos,
        para asegurar la reproducibilidad de los resultados. Por defecto es None.
    
    Returns:
    --------
    X_train : pandas.DataFrame
        Conjunto de datos de entrenamiento con las variables explicativas (independientes).
    
    X_test : pandas.DataFrame
        Conjunto de datos de prueba con las variables explicativas (independientes).
    
    y_train : pandas.Series
        Conjunto de datos de entrenamiento con la variable objetivo (dependiente).
    
    y_test : pandas.Series
        Conjunto de datos de prueba con la variable objetivo (dependiente).
    """
    # Separar variables explicativas de la variable objetivo
    X = df.drop(columns = [variable_objetivo]) #Variables explicativas
    y = df[variable_objetivo]
    
    # Dividir los datos en train y test
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size = test_size, random_state = random_state)
    
    # Separar las columnas numéricas y categóricas
    variables_num = X.select_dtypes(include = ['number']).columns.tolist()
    variables_cat = X.select_dtypes(exclude = ['number']).columns.tolist()
    
    # Convertir variables categóricas en dummies
    X_train = pd.get_dummies(X_train, columns = variables_cat, dtype = float)
    X_test = pd.get_dummies(X_test, columns = variables_cat, dtype = float)

    # X_train = pd.concat([X, cars_dummies], axis=1)
    # X_test = pd.concat([X, cars_dummies], axis=1)
    
    # Escalar variables numéricas
    scaler = StandardScaler()
    X_train[variables_num] = scaler.fit_transform(X_train[variables_num])
    X_test[variables_num] = scaler.transform(X_test[variables_num])
    
    return X_train, X_test, y_train, y_test

xx_train, xx_test, yy_train, yy_test = preparar_data(df, 'Price', 0.25, 42)

# Se aplica transformación polinomial a las variables explicativas
degree = 2  # Grado del polinomio (puedes ajustar este valor)
poly_features = PolynomialFeatures(degree = degree)
xx_train_poly = poly_features.fit_transform(xx_train)
xx_test_poly = poly_features.transform(xx_test)

# Se definen los valores de alpha que se probarán
alphas = [0.01, 0.1, 1, 10, 100]

# Se crea un diccionario con los hiperparámetros a ajustar
param_grid = {'alpha': alphas}

# Se crea un modelo Ridge
ridge_model = Ridge()

# Se realiza una búsqueda en cuadrícula para encontrar los mejores hiperparámetros
grid_search = GridSearchCV(ridge_model, param_grid, cv = 5, scoring = 'neg_mean_squared_error')
grid_search.fit(xx_train_poly, yy_train)

# Se obtiene los mejores hiperparámetros encontrados
best_alpha = grid_search.best_params_['alpha']

# Se entrena un modelo Ridge con los mejores hiperparámetros
ridge_model = Ridge(alpha = best_alpha)
ridge_model.fit(xx_train_poly, yy_train)

# Se realiza predicciones en el conjunto de prueba
yy_test_pred = ridge_model.predict(xx_test_poly)

# Se calcula RMSE para el conjunto de prueba
rmse_test = np.sqrt(mean_squared_error(yy_test, yy_test_pred))

# Se calcula R-squared para el conjunto de prueba
r2_test = r2_score(yy_test, yy_test_pred)

# Se calcula el R-squared ajustado para el conjunto de prueba
n = len(yy_test)
p = xx_test_poly.shape[1]
adjusted_r2_test = 1 - (1 - r2_test) * (n - 1) / (n - p - 1)

# Se calcula el Error Absoluto Medio (MAE) para el conjunto de prueba
mae_test = mean_absolute_error(yy_test, yy_test_pred)

# Se imprime métricas de evaluación
print(f'RMSE (Prueba): {rmse_test}')
print(f'R-squared (Prueba): {r2_test}')
print(f'R-squared ajustado (Prueba): {adjusted_r2_test}')
print(f'Error Absoluto Medio (MAE) (Prueba): {mae_test}')
